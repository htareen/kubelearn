Get a shell to the running container:
========================================

create shell-demo.yaml as below 
apiVersion: v1
kind: Pod
metadata:
  name: shell-demo
spec:
  volumes:
  - name: shared-data
    emptyDir: {}
  containers:
  - name: nginx
    image: nginx
    volumeMounts:
    - name: shared-data
      mountPath: /usr/share/nginx/html
  hostNetwork: true
  dnsPolicy: Default

kubectl apply -f https://k8s.io/examples/application/shell-demo.yaml
kubectl exec --stdin --tty shell-demo -- /bin/bash
root@ubuntu-1:~# microk8s.kubectl run --generator run-pod/v1 tmp-shell --rm -it --image bretfisher/netshoot -- bash
